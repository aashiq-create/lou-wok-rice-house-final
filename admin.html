require('dotenv').config(); const express = require('express'); const crypto = require('crypto'); const nodemailer = require('nodemailer'); const { google } = require('googleapis'); const path = require('path'); const app = express(); app.use(express.json()); const ADMIN_TOKEN = process.env.ADMIN_SECRET_TOKEN; const resetCodes = new Map(); // ─── Secret URL middleware ─────────────────────────────────────────── function requireAdminToken(req, res, next) { const token = req.params.token || req.headers['x-admin-token']; if (!token || token !== ADMIN_TOKEN) { return res.status(404).send('Not found'); // hard 404 — reveals nothing } next(); } // ─── Admin panel route (secret URL) ────────────────────────────────── app.get('/admin/:token', requireAdminToken, (req, res) => { res.sendFile(path.join(__dirname, 'public', 'admin.html')); }); // ─── Gmail transporter ─────────────────────────────────────────────── async function getTransporter() { const oauth2Client = new google.auth.OAuth2( process.env.GMAIL_CLIENT_ID, process.env.GMAIL_CLIENT_SECRET, 'https://developers.google.com/oauthplayground' ); oauth2Client.setCredentials({ refresh_token: process.env.GMAIL_REFRESH_TOKEN }); const { token } = await oauth2Client.getAccessToken(); return nodemailer.createTransport({ service: 'gmail', auth: { type: 'OAuth2', user: process.env.GMAIL_USER, clientId: process.env.GMAIL_CLIENT_ID, clientSecret: process.env.GMAIL_CLIENT_SECRET, refreshToken: process.env.GMAIL_REFRESH_TOKEN, accessToken: token, }, }); } // ─── Password reset routes ──────────────────────────────────────────── app.post('/auth/request-reset', async (req, res) => { const { email } = req.body; if (!email) return res.status(400).json({ error: 'Email required' }); const code = crypto.randomInt(100000, 999999).toString(); resetCodes.set(email, { code, expiry: Date.now() + 15 * 60 * 1000 }); try { const transporter = await getTransporter(); await transporter.sendMail({ from: `"${process.env.SITE_NAME || 'Admin'}" <${process.env.GMAIL_USER}>`, to: email, subject: 'Your password reset code', html: `

Reset your password

Your 6-digit code is:

${code}

Expires in 15 minutes. If you didn't request this, ignore this email.

`, }); res.json({ success: true }); } catch (err) { console.error('Gmail error:', err); res.status(500).json({ error: 'Failed to send reset email' }); } }); app.post('/auth/verify-code', (req, res) => { const { email, code } = req.body; const record = resetCodes.get(email); if (!record || record.code !== code || Date.now() > record.expiry) { return res.status(400).json({ error: 'Invalid or expired code' }); } res.json({ success: true }); }); app.post('/auth/verify-reset', (req, res) => { const { email, code, newPassword } = req.body; const record = resetCodes.get(email); if (!record || record.code !== code || Date.now() > record.expiry) { return res.status(400).json({ error: 'Invalid or expired code' }); } if (!newPassword || newPassword.length < 8) { return res.status(400).json({ error: 'Password too short' }); } // TODO: hash with bcrypt and save to your DB here resetCodes.delete(email); res.json({ success: true }); }); // ─── Deploy route (secret token required) ──────────────────────────── app.post('/admin/deploy', requireAdminToken, async (req, res) => { try { const response = await fetch(process.env.VERCEL_DEPLOY_HOOK, { method: 'POST' }); const data = await response.json(); res.json({ success: true, deploymentId: data.job?.id }); } catch (err) { console.error('Deploy error:', err); res.status(500).json({ error: 'Deploy failed' }); } }); // ─── Deploy status polling ──────────────────────────────────────────── app.get('/admin/deploy-status/:id', requireAdminToken, async (req, res) => { try { const r = await fetch( `https://api.vercel.com/v13/deployments/${req.params.id}`, { headers: { Authorization: `Bearer ${process.env.VERCEL_TOKEN}` } } ); const data = await r.json(); res.json({ state: data.readyState, url: data.url }); } catch (err) { res.status(500).json({ error: 'Status check failed' }); } }); app.listen(process.env.PORT || 3000, () => { console.log(`Server running. Admin panel at /admin/${ADMIN_TOKEN}`); });
